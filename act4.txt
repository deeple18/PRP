import java.util.*;
import java.lang.*;

public class act4 {
    public static void main(String[] args) {
        Scanner sc=new Scanner(System.in);
        int n,i,fact=1;
        System.out.print("Enter the value:");
        n=sc.nextInt();
        for(i=1;i<=n;i++)
        {
            fact=fact*i;
        }
        System.out.println("Factorial of "+n+" is "+fact);
    }
    
}
